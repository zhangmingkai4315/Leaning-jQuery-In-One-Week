exec永远返回与第一个匹配相关的信息，其返回数组包括第一个匹配的字串，所有分组的反向引用。

match是返回所有匹配的字符串合成的数组，但是正则表达式必须指定全局g属性才能返回所有匹配，不指定g属性则会返回一个只有一个元素的数组。

var text="Ubuntu 10";
var reg1=/^[a-z]+\s+\d+$/i;
var value1=reg1.exec(text);

var text2="Ubuntu 10.04";
var reg2=/^[a-z]+\s+(\d+)\.(\d+)$/i;
var value2=reg2.exec(text2);
$("#answer2").html(value2[0]);
//value[1]=10 value[2]=04

//惰性与贪婪捕获
var text3="abcdefg";
var reg3=/\w+/;
var reg3_lazy=/\w+?/;
var value3=reg3.exec(text3);
var value3_lazy=reg3_lazy.exec(text3);
$("#answer3").html("Default:"+value3+" Lazy:"+value3_lazy);

//捕获型分组与非捕获性分组
var text4="1234 21";
var reg4=/(\d+)+\s+(\d+)/;
var value4=text4.replace(reg4,"$2 $1");
$("#answer4").html(value4);
